def solution(data, col, row_begin, row_end):
    answer = 0
    data.sort(key=lambda x: (x[col-1], -x[0]))
    s = []
    for i in range(row_begin-1, row_end):
        s.append(sum(map(lambda x: x%(i+1), data[i]))) 
    
    for i in s:
        answer ^= i
    return answer
