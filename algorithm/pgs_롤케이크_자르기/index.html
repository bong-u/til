def solution(topping):
    answer = 0
    length = len(topping)
    toppingA = []
    toppingB = []
    setA = set()
    setB = set()

    for i in range(length-1):
        setA.add(topping[i])
        setB.add(topping[length-i-1])
        toppingA.append(len(setA))
        toppingB.append(len(setB))

    toppingB = toppingB[::-1]
    for i in range(length-1):
        if toppingA[i] == toppingB[i]:
            answer+=1
    return answer
