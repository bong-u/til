 3vx = [0, 0, -1, 1]
 4vy = [-1, 1, 0, 0]

 6def bfs(maps, N, M, p1, p2):
 7    visited = [[False]*M for _ in range(N)]
 8    q = deque()
 9    q.append(list(p1)+[0])
10
11    while q:
12        curY, curX, cnt = q.popleft()
13
14        if p2 == (curY, curX):
15            return cnt
16
17        for i in range(4):
18            x = curX + vx[i]
19            y = curY + vy[i]
20            if 0 <= x < M and 0 <= y < N and not visited[y][x] and maps[y][x] != 'X':
21                visited[y][x] = True
22                q.append([y, x, cnt+1])
23    return -1

30    sp = (0, 0)

34    for i in range(N):
