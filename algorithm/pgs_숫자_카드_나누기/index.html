    import math

def gcd(a, b):
    while b > 0:
        a, b = b, a%b
    return a

def gcdOfArr(l):
    result = l[0]
    for i in range(1, len(l)):
        result = gcd(result, l[i])
    return result

def solution(arrayA, arrayB):
    a1 = gcdOfArr(arrayA)
    for i in arrayB:
        if i % a1 == 0:
            a1 = 0
            break
    a2 = gcdOfArr(arrayB)
    for i in arrayA:
        if i % a2 == 0:
            a2 = 0
            break
    return max(a1, a2)

    import math
from functools import reduce

def gcd(a, b):
    while b > 0:
        a, b = b, a%b
    return a

def solution(arrayA, arrayB):
    a1 = reduce(gcd, arrayA)
    a1 = 0 if any(i % a1 == 0 for i in arrayB) else a1
    a2 = reduce(gcd, arrayB)
    a2 = 0 if any(i % a2 == 0 for i in arrayA) else a2
    return max(a1, a2)
