 1N, M = map(int, input().split())
 2
 3L = [list(map(int, input())) for _ in range(N)]
 4
 5dp = [[0]*(M+1) for _ in range(N+1)]
 6
 7result = 0
 8
 9for i in range(1, N+1):
10    for j in range(1, M+1):
11        dp[i][j] = L[i-1][j-1]

13        if L[i-1][j-1]:
14            dp[i][j] = min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1])+1
