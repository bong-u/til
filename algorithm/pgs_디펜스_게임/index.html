import heapq

def solution(n, k, enemy):
    heap = []
    cnt = 0 
    for i in enemy:
        n -= i
        heapq.heappush(heap, -i)
        while n < 0:
            k -= 1
            if not heap or k < 0:
                return cnt
            tmp = -heapq.heappop(heap)
            n += tmp
        cnt += 1
    return cnt
