import math

def diff(start, end):
    sh, sm = map(int, start.split(':'))
    eh, em = map(int, end.split(':'))
    
    return (eh*60+em) - (sh*60+sm)

def calc(time, baseTime, baseFee, unitTime, unitFee):
    if time < baseTime:
        return baseFee
    time -= baseTime
    return baseFee + math.ceil(time/unitTime)*unitFee
